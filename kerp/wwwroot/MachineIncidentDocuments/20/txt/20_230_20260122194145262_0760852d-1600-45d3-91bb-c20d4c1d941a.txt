using kerp.Contexts;
using kerp.Enums;
using kerp.Prosedur.MachineIncident.Event;
using kerp.Prosedur.MachineIncident.Incident;
using kerp.Prosedur.MachineIncident.MachineIncidentWorkShift;

using kerp.Prosedur.MachineIncident.Type;
using kerp.Prosedur.MachineIncident.WorkOrderType;
using Microsoft.EntityFrameworkCore;

namespace kerp.Repository.MachineIncidentRepository
{
    public class MachineIncidentRepository : IMachineIncidentRepository
    {
        private readonly KerpContext _ctx;

        public MachineIncidentRepository(KerpContext ctx)
        {
            _ctx = ctx;
        }

        #region Select Multi

        public List<MachineIncidentCrashTypeSelectMulti>? MachineIncidentCrashTypeSelectMulti()
        {
            return ExecuteList<MachineIncidentCrashTypeSelectMulti>("EXEC dbo.MachineIncidentCrashTypeSelectMulti");
        }

        public List<MachineIncidentProjectSelectMulti>? MachineIncidentProjectSelectMulti()
        {
            return ExecuteList<MachineIncidentProjectSelectMulti>("EXEC dbo.MachineIncidentProjectSelectMulti");
        }

        public List<MachineIncidentWorkOrderTypeSelectMulti>? MachineIncidentWorkOrderTypeSelectMulti()
        {
            return ExecuteList<MachineIncidentWorkOrderTypeSelectMulti>("EXEC dbo.MachineIncidentWorkOrderTypeSelectMulti");
        }

        public List<MachineIncidentWorkShiftSelectMulti>? MachineIncidentWorkShiftSelectMulti()
        {
            return ExecuteList<MachineIncidentWorkShiftSelectMulti>("EXEC dbo.MachineIncidentWorkShiftSelectMulti");
        }

        #endregion

        #region Post

        public int Post(List<MachineIncidentInsert> inserts)
        {
            using Microsoft.EntityFrameworkCore.Storage.IDbContextTransaction transaction = _ctx.Database.BeginTransaction();

            try
            {
                foreach (MachineIncidentInsert item in inserts)
                {
                    MachineIncidentSelectForBackEnd incident = InsertMachineIncident(item);
                    if (incident == null)
                    {
                        throw new Exception("MachineIncidentInsert failed");
                    }

                    InsertCrashType(incident.Id, item);
                    InsertWorkOrderType(incident.Id, item);
                    InsertSection(incident.Id, item);
                    InsertStructure(incident.Id, item);
                    InsertWorkShift(incident.Id, item);
                    InsertCreatedEvent(incident.Id, item.UserId);
                    InsertAssignedEvent(incident.Id, item.UserId, item.AsigntUserId);
                }

                transaction.Commit();
                return 1;
            }
            catch
            {
                transaction.Rollback();
                return 0;
            }
        }

        #endregion

        #region Insert Methods

        private MachineIncidentSelectForBackEnd InsertMachineIncident(MachineIncidentInsert item)
        {
            return ExecuteSingle<MachineIncidentSelectForBackEnd>(
                        "EXEC dbo.MachineIncidentInsert @p0, @p1, @p2, @p3",
                        item.Title,
                        item.UserId,
                        item.AssetId,
                        item.ProjectId
                    )!;
        }

        private void InsertCrashType(int incidentId, MachineIncidentInsert item)
        {
            _ = ExecuteSingle<MachineIncidentCrashTypeSelect>(
                        "EXEC dbo.MachineIncidentCrashTypeInsert @p0, @p1, @p2",
                        incidentId,
                        item.UserId,
                        item.MachineIncidentCrashTypeInsert!.CrashTypeId
                    );
        }

        private void InsertWorkOrderType(int incidentId, MachineIncidentInsert item)
        {
            _ = ExecuteSingle<MachineIncidentWorkOrderTypeSelect>(
                        "EXEC dbo.MachineIncidentWorkOrderTypeInsert @p0, @p1, @p2",
                        incidentId,
                        item.UserId,
                        item.MachineIncidentWorkOrderTypeInsert!.WorkOrderTypeId
                    );
        }

        private void InsertSection(int incidentId, MachineIncidentInsert item)
        {
            _ = ExecuteSingle<MachineIncidentSectionSelect>(
                        "EXEC dbo.MachineIncidentSectionInsert @p0, @p1, @p2",
                        incidentId,
                        item.UserId,
                        item.MachineIncidentSectionInsert!.SectionId
                    );
        }

        private void InsertStructure(int incidentId, MachineIncidentInsert item)
        {
            _ = ExecuteSingle<MachineIncidentStructureSelect>(
                        "EXEC dbo.MachineIncidentStructureInsert @p0, @p1, @p2",
                        incidentId,
                        item.UserId,
                        item.MachineIncidentStructureInsert!.StructureId
                    );
        }

        private void InsertWorkShift(int incidentId, MachineIncidentInsert item)
        {
            _ = ExecuteSingle<MachineIncidentWorkShiftSelect>(
                        "EXEC dbo.MachineIncidentWorkShiftInsert @p0, @p1, @p2",
                        incidentId,
                        item.UserId,
                        item.MachineIncidentWorkShiftInsert!.WorkShiftId
                    );
        }

        #endregion

        #region Event Inserts

        private void InsertCreatedEvent(int incidentId, int userId)
        {
            InsertEvent(incidentId, userId, MachineIncidentEventType.Created, userId);
        }

        private void InsertAssignedEvent(int incidentId, int userId, int assignUserId)
        {
            InsertEvent(incidentId, userId, MachineIncidentEventType.Assigned, assignUserId);
        }

        private void InsertEvent(
            int incidentId,
            int userId,
            MachineIncidentEventType eventType,
            int targetUserId)
        {
            _ = ExecuteSingle<MachineIncidentEventSelect>(
                "EXEC dbo.MachineIncidentEventInsert @p0, @p1, @p2, @p3",
                incidentId,
                userId,
                (int)eventType,
                targetUserId
            );
        }

        #endregion

        #region Helpers

        private List<T>? ExecuteList<T>(string sql) where T : class
        {
            return _ctx.Set<T>()
                        .FromSqlRaw(sql)
                        .AsNoTracking()
                        .ToList();
        }

        private T? ExecuteSingle<T>(string sql, params object[] parameters) where T : class
        {
            return _ctx.Set<T>()
                        .FromSqlRaw(sql, parameters)
                        .AsNoTracking()
                        .FirstOrDefault();
        }

        #endregion
    }
}

class SessionManager {
  static AuthSessionModel? _session;
  static UserModel? _user;
  static final ValueNotifier<int> userPageVersion = ValueNotifier<int>(0);

  static final BaseHiveStore<AuthSessionModel> _authStore = _AuthStore();
  static final BaseHiveStore<UserModel> _userStore = _UserStore();

  // 🔹 Login history üçün ayrıca store
  static final BaseHiveStore<LoginHistoryModel> _loginHistoryStore =
      _LoginHistoryStore();

  static AuthSessionModel? get session => _session;
  static UserModel? get user => _user;

  static bool get hasValidSession {
    if (_session == null) return false;
    return _session!.expiresAt.isAfter(DateTime.now());
  }

  static Future<void> loadFromHive() async {
    final authAll = await _authStore.getAll();
    if (authAll.isNotEmpty) {
      _session = authAll.first;
    }

    final userAll = await _userStore.getAll();
    if (userAll.isNotEmpty) {
      _user = userAll.first;
    }
  }

  static Future<void> saveSession({
    required AuthSessionModel session,
    required UserModel user,
  }) async {
    _session = session;
    _user = user;

    await _authStore.clear();
    await _userStore.clear();

    await _authStore.add(session);
    await _userStore.add(user);
  }

  static Future<void> clearSession() async {
    _session = null;
    _user = null;
    await _authStore.clear();
    await _userStore.clear();
    await Get.find<PageHubService>().stop();
  }

  static Future<void> logoutAndRedirectToLogin() async {
    await clearSession();
    if (Get.currentRoute != loginRoute) {
      Get.offAllNamed(loginRoute);
    }
  }

  // ================== LOGIN HISTORY ==================

  /// Uğurlu login olanda çağır: verilən username-i history-yə əlavə edir
  static Future<void> addLoginHistory(String username) async {
    final trimmed = username.trim();
    if (trimmed.isEmpty) return;

    final entry = LoginHistoryModel(
      username: trimmed,
      lastUsedAt: DateTime.now().toUtc(),
    );

    await _loginHistoryStore.add(entry);
  }

  /// Bütün unikal username-ləri son istifadə tarixinə görə qaytarır (ən yenilər əvvəl)
  static Future<List<String>> getAllLoginUsernames() async {
    final all = await _loginHistoryStore.getAll();

    final Map<String, LoginHistoryModel> latest = {};
    for (final item in all) {
      final existing = latest[item.username];
      if (existing == null || existing.lastUsedAt.isBefore(item.lastUsedAt)) {
        latest[item.username] = item;
      }
    }

    final list = latest.values.toList()
      ..sort((a, b) => b.lastUsedAt.compareTo(a.lastUsedAt));

    return list.map((e) => e.username).toList();
  }

  /// TextField üçün suggestion (yazılan string-ə görə filter)
  static Future<List<String>> searchLoginUsernames(
    String query, {
    int maxResults = 5,
  }) async {
    final trimmed = query.trim().toLowerCase();
    if (trimmed.isEmpty) return [];

    final all = await getAllLoginUsernames();

    return all
        .where((u) => u.toLowerCase().contains(trimmed))
        .take(maxResults)
        .toList();
  }

  // ================== USER PAGE UPDATE ==================

  /// Yalnız user-in içindəki userPage listini yenilə (başqa field-lərə toxunma)
  static Future<void> updateUserPages(List<UserPage> newPages) async {
    final current = _user;
    if (current == null) return;

    final updated = current.copyWith(userPage: newPages);

    _user = updated;

    await _userStore.clear();
    await _userStore.add(updated);
    userPageVersion.value++;
  }

  // ================== USER PROFILE UPDATE ==================

  static Future<void> updateUserProfile({
    String? firstName,
    String? lastName,
    String? position,
  }) async {
    final current = _user;
    if (current == null) return;

    final updated = current.copyWith(
      firstName: firstName,
      lastName: lastName,
      position: position,
    );

    _user = updated;

    await _userStore.clear();
    await _userStore.add(updated);
    // bu notifier-i profil üçün də istifadə edə bilərik
    userPageVersion.value++;
  }

  static Future<void> updateUserRelations({
    List<UserMailInfoModel>? userMailInfoSelect,
    List<UserPhoneInfoModel>? userPhoneInfoSelect,
    List<UserManagerEmployeModel>? userManagerEmployeSelect,
    List<UserUserConMachineModel>? userUserConMachineSelect,
    List<UserUserLoginsModel>? userUserLoginsSelect,
  }) async {
    final current = _user;
    if (current == null) return;

    final updated = current.copyWith(
      userMailInfoSelect: userMailInfoSelect,
      userPhoneInfoSelect: userPhoneInfoSelect,
      userManagerEmployeSelect: userManagerEmployeSelect,
      userUserConMachineSelect: userUserConMachineSelect,
      userUserLoginsSelect: userUserLoginsSelect,
    );

    _user = updated;
    await _userStore.clear();
    await _userStore.add(updated);
    userPageVersion.value++;
  }

  static Future<void> replaceUser(UserModel newUser) async {
    _user = newUser;
    await _userStore.clear();
    await _userStore.add(newUser);
    userPageVersion.value++;
  }
}

class ProjectManagementController extends GetxController {
  final GetAdminProjectsUseCase _getUseCase;
  final CreateProjectUseCase _createUseCase;
  final UpdateProjectUseCase _updateUseCase;
  final ChangeProjectStatusUseCase _changeStatusUseCase;

  ProjectManagementController(
    this._getUseCase,
    this._createUseCase,
    this._updateUseCase,
    this._changeStatusUseCase,
  );

  // Data
  final RxList<ProjectAdmin> items = <ProjectAdmin>[].obs;
  final RxBool loading = false.obs;
  final RxSet<int> statusLoading = <int>{}.obs;

  // Search
  final RxString searchQuery = ''.obs;

  // Pagination
  final int pageSize = 10;
  final RxInt currentPage = 0.obs;

  List<ProjectAdmin> get filteredItems {
    final q = searchQuery.value.trim().toLowerCase();
    if (q.isEmpty) return items.toList();

    return items.where((e) {
      final idStr = e.id.toString();
      final title = e.title.toLowerCase();
      return idStr.contains(q) || title.contains(q);
    }).toList();
  }

  int get totalPages {
    final list = filteredItems;
    if (list.isEmpty) return 0;
    return (list.length + pageSize - 1) ~/ pageSize;
  }

  List<ProjectAdmin> get pagedItems {
    final list = filteredItems;
    if (list.isEmpty) return const <ProjectAdmin>[];

    final start = currentPage.value * pageSize;
    if (start >= list.length) return const <ProjectAdmin>[];

    var end = start + pageSize;
    if (end > list.length) end = list.length;

    return list.sublist(start, end);
  }

  void setPage(int page) {
    if (page < 0) return;
    final maxPage = totalPages - 1;
    if (maxPage < 0) return;
    if (page > maxPage) return;
    currentPage.value = page;
  }

  void setSearchQuery(String value) {
    searchQuery.value = value;
    currentPage.value = 0;
  }

  @override
  void onInit() {
    super.onInit();
    fetchItems();
  }

  Future<void> fetchItems() async {
    loading.value = true;
    try {
      final list = await _getUseCase.execute();
      items.assignAll(list);
      currentPage.value = 0;
    } catch (e, s) {
      await LoggerService.uiError(
        error: e,
        stackTrace: s,
        route: Get.currentRoute,
        widget: 'ProjectManagementController.fetchItems',
      );
      SnackbarManager.show(
        SnackbarType.error,
        T.get(uiSnackbarLoadErrorKeys),
      );
    } finally {
      loading.value = false;
    }
  }

  bool isStatusLoading(int id) => statusLoading.contains(id);

  Future<void> changeStatus(ProjectAdmin item, bool activate) async {
    final userId = SessionManager.user?.userId;
    if (userId == null) {
      SnackbarManager.show(SnackbarType.error, T.get(generalKeys));
      return;
    }

    statusLoading.add(item.id);
    try {
      final updated = await _changeStatusUseCase.execute(
        id: item.id,
        userId: userId,
        activate: activate,
      );

      final index = items.indexWhere((e) => e.id == updated.id);
      if (index != -1) {
        items[index] = updated;
      } else {
        items.add(updated);
      }
      items.refresh();

      SnackbarManager.show(
        SnackbarType.success,
        T.get(uiSnackbarStatusChangeSuccessKeys),
      );
    } catch (e, s) {
      await LoggerService.uiError(
        error: e,
        stackTrace: s,
        route: Get.currentRoute,
        widget: 'ProjectManagementController.changeStatus',
      );
      SnackbarManager.show(
        SnackbarType.error,
        T.get(uiSnackbarStatusChangeErrorKeys),
      );
    } finally {
      statusLoading.remove(item.id);
    }
  }

  Future<void> createItem({required String title}) async {
    final userId = SessionManager.user?.userId;
    if (userId == null) {
      SnackbarManager.show(SnackbarType.error, T.get(generalKeys));
      return;
    }

    try {
      loading.value = true;

      final created = await _createUseCase.execute(
        userId: userId,
        title: title,
      );

      items.add(created);
      items.refresh();

      SnackbarManager.show(
        SnackbarType.success,
        T.get(uiSnackbarCreateSuccessKeys),
      );
    } catch (e, s) {
      await LoggerService.uiError(
        error: e,
        stackTrace: s,
        route: Get.currentRoute,
        widget: 'ProjectManagementController.createItem',
      );
      SnackbarManager.show(
        SnackbarType.error,
        T.get(uiSnackbarCreateErrorKeys),
      );
    } finally {
      loading.value = false;
    }
  }

  Future<void> updateItem({required int id, required String title}) async {
    final userId = SessionManager.user?.userId;
    if (userId == null) {
      SnackbarManager.show(SnackbarType.error, T.get(generalKeys));
      return;
    }

    try {
      loading.value = true;

      final updated = await _updateUseCase.execute(
        id: id,
        userId: userId,
        title: title,
      );

      final index = items.indexWhere((e) => e.id == updated.id);
      if (index != -1) {
        items[index] = updated;
      } else {
        items.add(updated);
      }
      items.refresh();

      SnackbarManager.show(
        SnackbarType.success,
        T.get(uiSnackbarUpdateSuccessKeys),
      );
    } catch (e, s) {
      await LoggerService.uiError(
        error: e,
        stackTrace: s,
        route: Get.currentRoute,
        widget: 'ProjectManagementController.updateItem',
      );
      SnackbarManager.show(
        SnackbarType.error,
        T.get(uiSnackbarUpdateErrorKeys),
      );
    } finally {
      loading.value = false;
    }
  }
}

class _ProjectPaginationBar extends StatelessWidget {
  final int currentPage;
  final int totalPages;
  final int pageSize;
  final int totalItems;
  final VoidCallback? onPrev;
  final VoidCallback? onNext;
  final void Function(int)? onPageSelected;

  const _ProjectPaginationBar({
    required this.currentPage,
    required this.totalPages,
    required this.pageSize,
    required this.totalItems,
    this.onPrev,
    this.onNext,
    this.onPageSelected,
  });

  @override
  Widget build(BuildContext context) {
    if (totalPages <= 1) return const SizedBox.shrink();

    final cs = Theme.of(context).colorScheme;
    final theme = Theme.of(context);

    final startIndex = totalItems == 0 ? 0 : currentPage * pageSize + 1;
    final rawEnd = (currentPage + 1) * pageSize;
    final endIndex = rawEnd > totalItems ? totalItems : rawEnd;

    final pages = List<int>.generate(totalPages, (i) => i);

    return Padding(
      padding: const EdgeInsets.fromLTRB(16, 0, 16, 12),
      child: Card(
        elevation: 0,
        color: cs.surfaceContainerHighest.withValues(alpha: .7),
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(999),
        ),
        child: Padding(
          padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 6),
          child: Row(
            children: [
              IconButton(
                icon: const Icon(Icons.chevron_left),
                onPressed: onPrev,
                tooltip: T.get(uiPaginationPrevTooltipKeys),
              ),
              const SizedBox(width: 4),
              Expanded(
                child: SingleChildScrollView(
                  scrollDirection: Axis.horizontal,
                  child: Row(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      for (final page in pages)
                        Padding(
                          padding: const EdgeInsets.symmetric(horizontal: 2),
                          child: ChoiceChip(
                            label: Text('${page + 1}'),
                            selected: page == currentPage,
                            onSelected: (_) => onPageSelected?.call(page),
                            selectedColor:
                                cs.primary.withValues(alpha: .16),
                          ),
                        ),
                    ],
                  ),
                ),
              ),
              const SizedBox(width: 4),
              IconButton(
                icon: const Icon(Icons.chevron_right),
                onPressed: onNext,
                tooltip: T.get(uiPaginationNextTooltipKeys),
              ),
              const SizedBox(width: 8),
              Text(
                totalItems == 0
                    ? '0 / 0'
                    : '$startIndex–$endIndex / $totalItems',
                style: theme.textTheme.bodySmall?.copyWith(
                  color: cs.onSurfaceVariant,
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

class _ProjectRow extends StatelessWidget {

        isActive ? T.get(uiStatusActiveKeys) : T.get(uiStatusPassiveKeys);

        isActive ? T.get(uiActionDeactivateKeys) : T.get(uiActionActivateKeys);

    return Card(
      elevation: 1,
      shadowColor: cs.shadow.withValues(alpha: .12),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
      child: Padding(
        padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 10),
        child: Row(

                color: cs.primary.withValues(alpha: .08),

                  style: theme.textTheme.titleSmall?.copyWith(

                    fontWeight: FontWeight.w700,
                    letterSpacing: 1.1,

              ),

            const SizedBox(width: 12),

            // Texts
            Expanded(
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [

                    children: [

                          item.title,

                            fontWeight: FontWeight.w600,

                    ),
                    child: Text(
                      statusText,
                      style: theme.textTheme.bodySmall?.copyWith(
                        color: statusColor,
                      ),
                    ),
                  ),
                ],
              ),

            // EDIT
            IconButton(
              tooltip: T.get(editKeys),
              onPressed: () => _openProjectDialog(context, editing: item),
              icon: const Icon(Icons.edit_outlined, size: 20),
            ),

            // ACTIVATE / DEACTIVATE
            Obx(() {
              final busy = controller.isStatusLoading(item.id);

              return IconButton(
                tooltip: statusTooltip,
                onPressed: busy

                    : () {
                        if (isActive) {
                          _showConfirmDialog(
                            context,

                            message: T.get(uiConfirmDeactivateMessageKeys),
                            onConfirm: () {
                              Get.back();
                              controller.changeStatus(item, false);

                          _showConfirmDialog(
                            context,
                            title: T.get(uiActionActivateKeys),
                            message: T.get(uiConfirmActivateMessageKeys),
                            onConfirm: () {
                              Get.back();
                              controller.changeStatus(item, true);
                            },
                          );
                        }
                      },
                icon: busy
                    ? const SizedBox(
                        width: 20,
                        height: 20,
                        child: CircularProgressIndicator(strokeWidth: 2),
                      )
                    : Icon(statusIcon, size: 20),
              );
            }),
          ],
        ),
      ),
    );
  }
}

Future<void> _openProjectDialog(
  BuildContext context, {
  ProjectAdmin? editing,
}) async {
  final controller = Get.find<ProjectManagementController>();
  final bool isEdit = editing != null;

  final titleCtrl = TextEditingController(text: editing?.title ?? '');
  final RxBool saving = false.obs;

  final theme = Theme.of(context);
  final cs = theme.colorScheme;

  final dialogTitle =
      isEdit ? T.get(uiDialogEditTitleKeys) : T.get(uiDialogNewTitleKeys);

  await Get.dialog(
    AlertDialog(
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
      title: Text(dialogTitle),
      content: ConstrainedBox(
        constraints: const BoxConstraints(maxWidth: 420),
        child: Column(
          mainAxisSize: MainAxisSize.min,
          children: [
            TextField(
              controller: titleCtrl,
              decoration: InputDecoration(
                labelText: T.get(projectManagementDialogFieldTitleKeys),
              ),
            ),
            const SizedBox(height: 12),
            if (isEdit)
              Align(
                alignment: Alignment.centerLeft,
                child: Container(
                  padding:
                      const EdgeInsets.symmetric(horizontal: 8, vertical: 4),
                  decoration: BoxDecoration(
                    color: (editing.status ? cs.primary : cs.error)
                        .withValues(alpha: .08),
                    borderRadius: BorderRadius.circular(999),
                  ),
                  child: Text(
                    editing.status
                        ? T.get(uiStatusActiveKeys)
                        : T.get(uiStatusPassiveKeys),
                    style: theme.textTheme.bodySmall?.copyWith(
                      color: editing.status ? cs.primary : cs.error,
                    ),
                  ),
                ),
              ),
          ],
        ),
      ),
      actions: [
        TextButton(
          onPressed: () => Get.back(),
          child: Text(T.get(cancelKeys)),
        ),
        Obx(
          () => ElevatedButton(
            onPressed: saving.value
                ? null
                : () async {
                    final titleText = titleCtrl.text.trim();

                    if (titleText.isEmpty) {
                      SnackbarManager.show(
                        SnackbarType.error,
                        T.get(uiValidationRequiredKeys),
                      );
                      return;
                    }

                    saving.value = true;

                    try {
                      if (isEdit) {
                        await controller.updateItem(
                          id: editing.id,
                          title: titleText,
                        );
                      } else {
                        await controller.createItem(title: titleText);
                      }
                    } catch (e, s) {
                      await LoggerService.uiError(
                        error: e,
                        stackTrace: s,
                        route: Get.currentRoute,
                        widget: '_openProjectDialog',
                      );
                      SnackbarManager.show(
                        SnackbarType.error,
                        T.get(generalKeys),
                      );
                    } finally {
                      saving.value = false;
                    }
                  },
            child: saving.value
                ? const SizedBox(
                    width: 18,
                    height: 18,
                    child: CircularProgressIndicator(strokeWidth: 2),
                  )
                : Text(T.get(saveKeys)),
          ),
        ),
      ],
    ),
  );
}

Future<void> _showConfirmDialog(
  BuildContext context, {
  required String title,
  required String message,
  required VoidCallback onConfirm,
}) async {
  await Get.dialog(
    AlertDialog(
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(14)),
      title: Text(title),
      content: Text(message),
      actions: [
        TextButton(
          onPressed: () => Get.back(),
          child: Text(T.get(noKeys)),
        ),
        ElevatedButton(
          onPressed: onConfirm,
          child: Text(T.get(yesKeys)),
        ),
      ],
    ),
  );
}
